Googleスプレッドシートへの自動送信

GASをデプロイしてURLを取得後、下に貼り付けて「送信」ボタンを押してください。

1
スプレッドシートのシート名を 「データ蓄積」 にする。
2
「拡張機能」→「Apps Script」→下のコードを貼り付けて保存。
3
「デプロイ」→「新しいデプロイ」→ウェブアプリ→アクセス権「全員」で公開。URLを下に貼る。
function doPost(e) {
  const sheet = SpreadsheetApp
    .getActiveSpreadsheet()
    .getSheetByName("データ蓄積");

  if (sheet.getLastRow() === 0) {
    sheet.appendRow([
      "日付","部門","No.","業務内容",
      "担当者","枚数","1枚あたり(分)",
      "合計時間(分)","備考"
    ]);
  }

  const rows = JSON.parse(e.postData.contents);
  rows.forEach(r => {
    sheet.appendRow([
      r.date, r.dept, r.no, r.task,
      r.person, r.count, r.per,
      r.total, r.note
    ]);
  });

  return ContentService
    .createTextOutput(JSON.stringify({status:"ok"}))
    .setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) {
  return ContentService
    .createTextOutput(JSON.stringify({status:"ok"}))
    .setMimeType(ContentService.MimeType.JSON);
}
